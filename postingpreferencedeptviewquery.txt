OUTER LOOP1---(jo_name,jo_code,station_pref,jo_remarks)

select station_name,remarks from judicial_officer_posting_preferences where final_submission='Y' and judicial_officer_id=1

 LOOP2---(posted as)

select jo.officer_name,designations.designation_name,jo_posting.from_date from judicial_officers as jo
left join judicial_officer_postings as jo_posting on jo.id=jo_posting.judicial_officer_id
join designations on jo_posting.designation_id=designations.id where jo.id=9 order by jo_posting.from_date desc limit 1

-------------------------------------------------------------------------------------------------------------------------------

 LOOP3---(practice subdivision)

select jo.officer_name,subdivisions.subdivision_name,jle.subdivision_id 
from 
judicial_officers as jo join jo_legal_experiences as jle on jle.judicial_officer_id=jo.id
left join subdivisions on jle.subdivision_id=subdivisions.id where jo.id=9 

-------------------------------------------------------------------------------------------------------------------------------

 LOOP4---(FETCHING THE SPOUSE NAME)
select jo1.id,jo1.officer_name as spouse_name
from judicial_officers as jo join judicial_officers as jo1 on jo.id=jo1.spouse 
where jo.id=1

------------------------------------------------------------------------------------------------------------------------------
